00054     for (j = 0 ; j < n ; j++)

00057         for (p = 0 ; p < len ; p++)

00066     for (j = 0 ; j < n ; j++)
00067     {
00068         W [j] = 0 ;
00069     }
00070     for (i = 0 ; i < n ; i++)
00071     {
00072         pend = Tp [i+1] ;
00073         for (p = Tp [i] ; p < pend ; p++)
00074         {
00075             j = Tj [p] ;
00076             GET_POINTER (LU, Xip, Xlen, Xi, Xx, j, len) ;
00077             xlen = W [j]++ ;
00078             Xi [xlen] = i ;
00079             Xx [xlen] = Tx [p] ;
00080         }
00081     }
00082 
00083     ASSERT (KLU_valid_LU (n, FALSE, Xip, Xlen, LU)) ;
00084 }
