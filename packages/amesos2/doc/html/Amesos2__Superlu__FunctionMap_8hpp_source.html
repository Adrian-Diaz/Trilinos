00050 template <typename Scalar>
00051 Teuchos::RCP<SLU::SuperMatrix> getPermMatrix(
00052   SLU::superlu_options_t* options,
00053   SLU::SuperMatrix* A,          // A is in NRFormat
00054   int* perm_c,
00055   int* etree)
00056 {
00057   SLU::SuperMatrix AA;      // A in SLU_NC format
00058 
00059   // AA in SLU_NCP format after preordering
00060   Teuchos::RCP<SLU::SuperMatrix> AC = Teuchos::rcp(new SLU::SuperMatrix());
00061 
00062   if( A->Stype == SLU::SLU_NR ){
00063     SLU::NRformat* Astore = (SLU::NRformat*)A->Store;
00064     // SLU::S::sCreate_CompCol_Matrix(&AA, A->ncol, A->nrow, Astore->nnz,
00065     //   (typename TypeMap<Amesos::Superlu,Scalar>::type*)Astore->nzval,
00066     //   Astore->colind, Astore->rowptr, SLU::SLU_NC, A->Dtype, A->Mtype);
00067     FunctionMap<Superlu,Scalar>::create_CompCol_Matrix(&AA, A->ncol, A->nrow,
00068       Astore->nnz, (typename TypeMap<Amesos::Superlu,Scalar>::type*)Astore->nzval,
00069       Astore->colind, Astore->rowptr, SLU::SLU_NC, A->Dtype, A->Mtype);
00070   } else {
00071     AA = *A;
00072   }
00073 
00074   /*
00075    * Get column permutation vector perm_c[], according to permc_spec:
00076    *   permc_spec = NATURAL:  natural ordering 
00077    *   permc_spec = MMD_AT_PLUS_A: minimum degree on structure of A'+A
00078    *   permc_spec = MMD_ATA:  minimum degree on structure of A'*A
00079    *   permc_spec = COLAMD:   approximate minimum degree column ordering
00080    *   permc_spec = MY_PERMC: the ordering already supplied in perm_c[]
00081    */
00082   int permc_spec = options->ColPerm;
00083   if ( permc_spec != SLU::MY_PERMC ){
00084     SLU::get_perm_c(permc_spec, &AA, perm_c);
00085   }
00086   SLU::sp_preorder(options, &AA, perm_c, etree, AC.getRawPtr());
00087 
00088   // cleanup
00089   SLU::Destroy_SuperMatrix_Store(&AA);
00090 
00091   return(AC);
00092 }

00096 /** 
00097  * Helper class which passes on function calls to the appropriate Superlu
00098  * function based on the type of its scalar template argument.
00099  *
00100  * Superlu has solver and matrix builder functions defined based on data type.
00101  * One function for complex, one for double precision complex, another for \c
00102  * float , and yet another for \c double.  To work elegantly with the
00103  * Amesos::Superlu interface we want to be able to perform a single function
00104  * call which is appropriate for the scalar type of the Matrix and
00105  * MultiVectors that we are working with.  The \c FunctionMap class
00106  * provides that capability.
00107  *
00108  * The class template is specialized for each data type that Superlu supports,
00109  * and errors are thrown for other data types.
00110  */
